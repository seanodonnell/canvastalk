Images and Animation

var canvas = document.getElementById("mycanvas");
var dlogo = new Image();
dlogo.src = './imgs/dwlogo.gif';
var dmovie = document.getElementById("dmovie");
dmovie.loop = true;
dmovie.play();
var context = canvas.getContext('2d');
var x = 0;
function draw()
{
    context.setTransform(1, 0, 0, 1, 0, 0);
    context.save();
    context.setTransform(1, 0, .1, 1, 0, 0);
    context.drawImage(dmovie, 20, 0);
    context.restore();
    context.drawImage(dlogo, x, 200,100, 100);
    x++;
    if(x>720)
    {
        x = -200;
    }
}
setInterval(draw, 10);
