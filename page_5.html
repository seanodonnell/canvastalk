Images and Animation

var canvas = document.getElementById("mycanvas");
var dlogo = new Image();
dlogo.src = './imgs/dwlogo.gif';
var rotation = 0;
function draw()
{
    canvas.width = canvas.width;
    var context = canvas.getContext('2d');
    context.translate(100, 100);
    context.rotate(rotation * Math.PI / 64);
    context.translate(-100, -100);
    context.drawImage(dlogo, 100-dlogo.width/2, 100-dlogo.height/2,
                  dlogo.width, dlogo.height);
    rotation++;
}
setInterval(draw,10);
